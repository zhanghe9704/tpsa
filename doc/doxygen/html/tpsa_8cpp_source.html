94 unsigned int comb_num(unsigned int n, unsigned int r)
95 {
96  const unsigned int MAX_N_R = 100;
97 
98  if (n == 0 || r == 0) return 1;
99 
100  // not allowed, regardless of mathematical meaning.
101  if (n < r) return 0;
102 
103  // obviously too large fitting in a unsigned int
104  if (r > MAX_N_R && (n-r) > MAX_N_R) return 0;
105 
106  unsigned int k = (r > n-r ? n-r : r);
107 
108  std::vector<int> numerator(k), denominator(k);
109 
110  for (unsigned int i = 0; i < k; ++i) {
111  numerator[i] = n - i;
112  denominator[i] = k - i;
113  }
114 
115  int c = 1;
116  for (size_t i = 0; i < k; ++i) {
117  // deal with denominator[i]
118  for (size_t j = 0; j < k; ++j) {
119  if ( (c=gcd(denominator[i], numerator[j])) > 1 ) {
120  if (numerator[j] % c != 0 || denominator[i] % c != 0) {
121  #ifdef DEBUG
122  std::cerr << "ERROR: gcd error, " << k << " = gcd("
123  << denominator[i] << ","
124  << numerator[j] << ")" << std::endl;
125  #endif
126 
127  return 0;
128  }
129  numerator[j] /= c;
130  denominator[i] /= c;
131  }
132  }
133  }
134 
135  // check if the result is an integer, i.e. the denominator vector is
136  // (1,1,1, ... ,1)
137  for (size_t i = 0; i < k; ++i) {
138  if (denominator[i] != 1) {
139  #ifdef DEBUG
140  std::cerr << "ERROR: denominator is not dividable by numerator"
141  << std::endl;
142  #endif
143 
144  return 0;
145  }
146  }
147 
148  size_t N = 1;
149 
150  for (size_t i = 0; i < k; ++i) {
151  if (MAX_N_BASE*1.0/numerator[i] < N) {
152  return 0;
153  }
154  N *= numerator[i];
155  }
156  return N;
157 }
