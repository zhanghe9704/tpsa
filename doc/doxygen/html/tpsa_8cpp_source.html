68unsigned int gcd(unsigned int a, unsigned int b)
69{
70 unsigned int t;
71 while (b != 0u) {
72 t = b;
73 b = a % b;
74 a = t;
75 }
76 return a;
77}

81// The result should fit into a unsigned int(e.g. on 32bit system 4294967296).
82//
83// C(15,10) = 3003
84// C(17,10) = 19448
85// C(18,10) = 43758
86// int16 = 65536
87// C(19,10) = 92378
88// C(20,10) = 184756
89// C(30,10) = 30045015
90// int32 = 4294967296
91unsigned int comb_num(unsigned int n, unsigned int r)
92{
93 const unsigned int MAX_N_R = 100;
94
95 if (n == 0 || r == 0) return 1;
96
97 // not allowed, regardless of mathematical meaning.
98 if (n < r) return 0;
99
100 // obviously too large fitting in a unsigned int
101 if (r > MAX_N_R && (n-r) > MAX_N_R) return 0;
102
103 unsigned int k = (r > n-r ? n-r : r);
104
105 std::vector<int> numerator(k), denominator(k);
106
107 for (unsigned int i = 0; i < k; ++i) {
108 numerator[i] = n - i;
109 denominator[i] = k - i;
110 }
111
112 int c = 1;
113 for (size_t i = 0; i < k; ++i) {
114 // deal with denominator[i]
115 for (size_t j = 0; j < k; ++j) {
116 if ( (c=gcd(denominator[i], numerator[j])) > 1 ) {
117 if (numerator[j] % c != 0 || denominator[i] % c != 0) {
118 #ifdef DEBUG
119 std::cerr << "ERROR: gcd error, " << k << " = gcd("
120 << denominator[i] << ","
121 << numerator[j] << ")" << std::endl;
122 #endif
123
124 return 0;
125 }
126 numerator[j] /= c;
127 denominator[i] /= c;
128 }
129 }
130 }
131
132 // check if the result is an integer, i.e. the denominator vector is
133 // (1,1,1, ... ,1)
134 for (size_t i = 0; i < k; ++i) {
135 if (denominator[i] != 1) {
136 #ifdef DEBUG
137 std::cerr << "ERROR: denominator is not dividable by numerator"
138 << std::endl;
139 #endif
140
141 return 0;
142 }
143 }
144
145 size_t N = 1;
146
147 for (size_t i = 0; i < k; ++i) {
148 if (MAX_N_BASE*1.0/numerator[i] < N) {
149 return 0;
150 }
151 N *= numerator[i];
152 }
153 return N;
154}

394extern "C" _declspec(dllexport) void _stdcall ad_init(const TNVND* nvar, const TNVND* nord)
395#else
396void ad_init(const TNVND* nvar, const TNVND* nord)
397#endif
398{
399 TNVND nv = *nvar;
400 TNVND nd = *nord;
401
402 #ifdef DEBUG
403 std::cerr << "Initialize nv= " << *nvar << " nd= " << *nord << std::endl;
404 #endif
405 advecpool = NULL;
406 gnv = nv; // assign to global nv
407 gnd = nd; // assign to global nd
408
409 FULL_VEC_LEN = comb_num(gnv+gnd, gnd);
410 init_order_index(gnv, gnd);
411
412 // overflow !!
413 if (FULL_VEC_LEN == 0) {
414 std::cerr << "Overflow! (" << nv << "," << nd << ")" << std::endl;
415 std::exit(1);
416 }
417
418 // this reset the base vector
419 vec = new TNVND[nv];
420 for (TNVND i = 0; i < nv; ++i) vec[i] = 0;
421
422 nvec = 0;
423
424 init_base(gnv, gnd);
425
426 init_prod_index(gnv, gnd);
427
428 // random number generator
429 srand(1U);
430}
