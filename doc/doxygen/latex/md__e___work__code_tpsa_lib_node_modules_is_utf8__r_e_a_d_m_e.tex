\#utf8 detector

Detect if a Buffer is utf8 encoded. It need The minimum amount of bytes is 4.


\begin{DoxyCode}{0}
\DoxyCodeLine{var fs = require('fs');}
\DoxyCodeLine{var isUtf8 = require('is-\/utf8');}
\DoxyCodeLine{var ansi = fs.readFileSync('ansi.txt');}
\DoxyCodeLine{var utf8 = fs.readFileSync('utf8.txt');}
\DoxyCodeLine{}
\DoxyCodeLine{console.log('ansi.txt is utf8: '+isUtf8(ansi)); //false}
\DoxyCodeLine{console.log('utf8.txt is utf8: '+isUtf8(utf8)); //true}

\end{DoxyCode}
