Chardet is a character detection module for Node\+JS written in pure Javascript. Module is based on ICU project \href{http://site.icu-project.org/}{\texttt{ http\+://site.\+icu-\/project.\+org/}}, which uses character occurency analysis to determine the most probable encoding.\hypertarget{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md771}{}\doxysection{Installation}\label{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md771}

\begin{DoxyCode}{0}
\DoxyCodeLine{npm i chardet}

\end{DoxyCode}
\hypertarget{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md772}{}\doxysection{Usage}\label{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md772}
To return the encoding with the highest confidence\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{var chardet = require('chardet');}
\DoxyCodeLine{chardet.detect(Buffer.alloc('hello there!'));}
\DoxyCodeLine{// or}
\DoxyCodeLine{chardet.detectFile('/path/to/file', function(err, encoding) \{\});}
\DoxyCodeLine{// or}
\DoxyCodeLine{chardet.detectFileSync('/path/to/file');}

\end{DoxyCode}


To return the full list of possible encodings\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{var chardet = require('chardet');}
\DoxyCodeLine{chardet.detectAll(Buffer.alloc('hello there!'));}
\DoxyCodeLine{// or}
\DoxyCodeLine{chardet.detectFileAll('/path/to/file', function(err, encoding) \{\});}
\DoxyCodeLine{// or}
\DoxyCodeLine{chardet.detectFileAllSync('/path/to/file');}
\DoxyCodeLine{}
\DoxyCodeLine{//Returned value is an array of objects sorted by confidence value in decending order}
\DoxyCodeLine{//e.g. [\{ confidence: 90, name: 'UTF-\/8'\}, \{confidence: 20, name: 'windows-\/1252', lang: 'fr'\}]}

\end{DoxyCode}
\hypertarget{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md773}{}\doxysection{Working with large data sets}\label{md__e___work__code_tpsa_lib_node_modules_chardet__r_e_a_d_m_e_autotoc_md773}
Sometimes, when data set is huge and you want to optimize performace (in tradeoff of less accuracy), you can sample only first N bytes of the buffer\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{chardet.detectFile('/path/to/file', \{ sampleSize: 32 \}, function(err, encoding) \{\});}

\end{DoxyCode}
